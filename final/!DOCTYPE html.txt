        function updateDashboard() {
            const enStock = stock.filter(v => v.Statut === 'En stock');
            const vendus = stock.filter(v => v.Statut === 'Vendu');
            const enReparation = stock.filter(v => v.Statut === 'En r√©paration');
            const reserves = stock.filter(v => v.Statut === 'R√©serv√©');
            
            // Mise √† jour des statistiques principales
            document.getElementById('total-stock-count').textContent = enStock.length;
            document.getElementById('total-sold-count').textContent = vendus.length;
            document.getElementById('total-stock-value').textContent = '$' + enStock.reduce((sum, v) => sum + v.Co√ªt, 0).toLocaleString();
            
            // Calcul des profits
            const profitVendus = vendus.reduce((sum, v) => sum + (v.PrixVente || 0 - v.Co√ªt), 0);
            document.getElementById('total-profit-value').textContent = '$' + profitVendus.toLocaleString();
            
            // Dashboard d√©taill√©
            document.getElementById('total-vehicles').textContent = stock.length;
            document.getElementById('total-cost').textContent = '$' + stock.reduce((sum, v) => sum + v.Co√ªt, 0).toLocaleString();
            document.getElementById('total-sales').textContent = '$' + vendus.reduce((sum, v) => sum + (v.PrixVente || 0), 0).toLocaleString();
            
            const avgProfit = vendus.length > 0 ? profitVendus / vendus.length : 0;
            document.getElementById('average-profit').textContent = '$' + avgProfit.toLocaleString();
        }
